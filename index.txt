<script>
let members = JSON.parse(localStorage.getItem('my_members')) || [];
let payments = JSON.parse(localStorage.getItem('my_payments')) || [];

function saveData() {
    localStorage.setItem('my_members', JSON.stringify(members));
    localStorage.setItem('my_payments', JSON.stringify(payments));
    renderAll();
}

function showPage(id, btn) {
    document.querySelectorAll('.page').forEach(p => p.classList.remove('active'));
    document.querySelectorAll('.nav-item').forEach(n => n.classList.remove('active'));
    document.getElementById('page-' + id).classList.add('active');
    btn.classList.add('active');
}

function openModal(id) { document.getElementById(id).classList.add('open'); }
function closeModal(id) { document.getElementById(id).classList.remove('open'); }

function addMember() {
    const name = document.getElementById('m-name').value;
    const phone = document.getElementById('m-phone').value;
    const fee = document.getElementById('m-fee').value;
    if(!name) return alert("İsim yazın");
    members.push({ id: Date.now(), name, phone, fee: Number(fee), debt: Number(fee) });
    saveData();
    closeModal('add-member-modal');
}

function addPayment() {
    const mid = document.getElementById('p-member').value;
    const amount = document.getElementById('p-amount').value;
    const member = members.find(m => m.id == mid);
    if(!member) return;
    member.debt -= Number(amount);
    payments.push({ id: Date.now(), memberName: member.name, amount: Number(amount), date: new Date().toLocaleDateString() });
    saveData();
    closeModal('add-payment-modal');
}

function deleteMember(id) {
    if(confirm('Üyeyi silmek istediğinize emin misiniz?')) {
        members = members.filter(m => m.id != id);
        saveData();
    }
}

function renderAll() {
    // Stats
    document.getElementById('stat-total').textContent = members.length;
    let totalPaid = payments.reduce((sum, p) => sum + p.amount, 0);
    let totalDebt = members.reduce((sum, m) => sum + m.debt, 0);
    document.getElementById('stat-money').textContent = '₺' + totalPaid;
    document.getElementById('stat-debt').textContent = '₺' + totalDebt;

    // Members List
    const mList = document.getElementById('members-list');
    mList.innerHTML = members.map(m => `
        <tr>
            <td><strong>${m.name}</strong></td>
            <td>${m.phone}</td>
            <td style="color:${m.debt > 0 ? 'red' : 'green'}">₺${m.debt}</td>
            <td><button class="btn" onclick="deleteMember(${m.id})">🗑️</button></td>
        </tr>
    `).join('');

    // Payments List
    const pList = document.getElementById('payments-list');
    pList.innerHTML = payments.map(p => `
        <tr><td>${p.memberName}</td><td>₺${p.amount}</td><td>${p.date}</td></tr>
    `).join('');

    // Payment Select
    const pSelect = document.getElementById('p-member');
    pSelect.innerHTML = members.map(m => `<option value="${m.id}">${m.name}</option>`).join('');
}

// İlk açılışta çalıştır
renderAll();
